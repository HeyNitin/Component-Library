    An alert displays a short, important message in a way that attracts the user's attention without interrupting the user's task.

    Alerts can be created using alert class, followed by the variant name, ie, use alert alt-primary for primary alert.

    Primary! This is a primary alert
    Success! This is a primary alert
    Warning! This is a primary alert
    Error! This is a primary alert 
    
          <div class="alert alt-primary jsalert">
          <a>Primary! This is a primary alert</a>
          <button> class="button jsbtn">×</button>
          </div>
          <div class="alert alt-success jsalert">
          <a>Success! This is a primary alert</a>
          <button> class="button jsbtn">×</button>
          </div>
          <div class="alert alt-warning jsalert">
          <a>Warning! This is a primary alert</a>
          <button> class="button jsbtn">×</button>
          </div>
          <div class="alert alt-error jsalert">
          <a>Error! This is a primary alert</a>
          <button> class="button jsbtn">×</button>
          </div>
      
    
        const btn = document.getElementsByClassName("jsbtn");
        const alert = document.getElementsByClassName("jsalert");
        const reset = document.getElementById("reset");

        for (let i = 0; i < btn.length; i++) {
        btn[i].onclick = () => (alert[i].style.display = "none");
        }

        reset.onclick = () => {
          for (let i = 0; i < btn.length; i++) {
            alert[i].style.display = "flex";
          }
        };
